# Vercel → Project → Settings → Environment Variables
#
# 로컬에서 `vercel dev` 로 같은 동작을 하려면, Vercel에 넣은 이름·값을
# `.env.local` 에 복사하면 됩니다. 틀만 보려면 저장소의 `.env.local.example` 을 참고하세요.

# 방법 A: 프로젝트 ref (xxx.supabase.co 의 xxx) + anon 키
SUPABASE_PROJECT_ID=your-project-ref
SUPABASE_ANON_KEY=your-anon-public-key

# 방법 B (권장): 전체 URL + anon 키
# SUPABASE_URL=https://xxxx.supabase.co
# SUPABASE_ANON_KEY=eyJhbGciOi...

# Vercel·Supabase 연동이 넣는 이름(프런트 노출용)도 /api/env 에서 읽습니다.
# NEXT_PUBLIC_SUPABASE_URL=...
# NEXT_PUBLIC_SUPABASE_ANON_KEY=...
# NEXT_PUBLIC_SUPABASE_PUBLISHABLE_OR_ANON_KEY=...  (신규 publishable 키일 때)

# 관리자 페이지(/admin) — API /api/admin/questions 에서만 사용. 임의 긴 문자열로 설정.
# OX_ADMIN_SECRET=...
# Vercel 프로덕션에도 SUPABASE_SERVICE_ROLE_KEY + OX_ADMIN_SECRET + (SUPABASE_URL 또는 PROJECT_ID) 필요
